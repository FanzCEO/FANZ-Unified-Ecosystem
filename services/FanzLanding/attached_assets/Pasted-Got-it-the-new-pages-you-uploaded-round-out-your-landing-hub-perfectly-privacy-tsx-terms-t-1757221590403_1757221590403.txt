2. Crawler

Create /client/scripts/crawlLinks.cjs to spider your site after build:

import fs from "node:fs";
import path from "node:path";
import { chromium } from "playwright";

const base = process.env.WEB_APP_URL || "http://localhost:3000";
const seen = new Set([base]);
const failures = [];

const browser = await chromium.launch();
const page = await browser.newPage();

while (seen.size) {
  const url = [...seen][0];
  seen.delete(url);

  try {
    const res = await page.goto(url, { waitUntil: "networkidle" });
    const status = res?.status?.() || 0;
    if (status >= 400 || status === 0) failures.push({ url, status });

    const links = await page.$$eval("a[href]", els => els.map(a => a.href));
    links.forEach(h => {
      if (h.startsWith(base)) {
        const clean = h.split("#")[0];
        if (!seen.has(clean)) seen.add(clean);
      }
    });
  } catch (e) {
    failures.push({ url, error: e.message });
  }
}

await browser.close();

const report = { checked: seen.size, failures };
fs.mkdirSync(path.resolve("client/dist"), { recursive: true });
fs.writeFileSync(path.resolve("client/dist/link-report.json"), JSON.stringify(report, null, 2));

if (failures.length) {
  console.error("‚ùå Crawl found issues:", failures);
  process.exit(1);
} else {
  console.log("‚úÖ Crawl passed ‚Äî no broken links");
}
